on_ruler_change = {

	on_actions = {
		on_monarchy_ruler_change
		delay = { days = { 1 2 } }
	}

}

on_monarchy_ruler_change = {
	trigger = {
		is_monarchy = yes
		NOT = { has_variable = regency_government }	
	}
	
	effect = {
		add_stability = -10
		calculate_succession_effect = yes
		trigger_event = {
			id = firstborn_traits.1
		}
	}
}

country_decisions = {
# Become Commander
	become_commander = {
		potential = {
			num_of_cities >= 1 # Required number of cities to take the decision.
			NOT = { has_variable = ruler_become_commander }
			OR = {
				has_law = work_for_the_best # Royal Army.
				has_law = standing_army_law # Professional Army.
			}
		}
		allow = {
			current_ruler = {
				is_male = yes
				is_adult = yes
			}
		}
		effect = {
			trigger_event = {
				id = commander_ruler_events.1
			}
		}
		ai_will_do = {
			base = 0 # The AI can't take the decision.
		}
	}

# Leave Command
	leave_command = {
		potential = {
			num_of_cities >= 1 # Required number of cities to take the decision.
			has_variable = ruler_become_commander 
		}
		allow = {
			current_ruler = {
				is_male = yes
			}
		}
		effect = {
			trigger_event = {
				id = commander_ruler_events.2
			}
		}
		ai_will_do = {
			base = 0 # The AI can't take the decision.
		}
	}

	fix_command = {
		potential = {
			num_of_cities >= 1 # Required number of cities to take the decision.
			any_character = {
				in_command = yes
				has_character_modifier = ruler_is_a_commander
				is_ruler = no
			}
		}
		allow = {
			current_ruler = {
				is_alive = yes
			}
		}
		effect = {
			trigger_event = {
				id = commander_ruler_events.5
			}
		}
		ai_will_do = {
			base = 0 # The AI can't take the decision.
		}
	}

	debug_ruler_test = {
		potential = {
			num_of_cities >= 1
		}
		allow = {
			current_ruler = {
				is_male = yes
			}
		}
		effect = {
			trigger_event = {
				id = commander_ruler_events.4
			}
		}

		ai_will_do = {
			base = 0
		}
	}
}
